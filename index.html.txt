<script>
  const sheetURL = "https://opensheet.elk.sh/1wxNP4hiaGLywX2g5NZi8jYVN7abF1sbf5DYSky97vZ0/Sheet1";

  function showModal(name, description) {
    document.getElementById('watch-name').textContent = name;
    document.getElementById('watch-desc').textContent = description;
    document.getElementById('modal').classList.add('active');
  }

  function closeModal() {
    document.getElementById('modal').classList.remove('active');
  }

  fetch(sheetURL)
    .then(response => response.json())
    .then(data => {
      const container = document.getElementById("tackymeter");

      data.forEach((watch, index) => {
        const watchDiv = document.createElement("div");
        watchDiv.className = "watch";

        // Horizontal position
        const position = parseFloat(watch["Position (0-100)"]);
        watchDiv.style.left = !isNaN(position) ? `${position}%` : "0%";

        // Vertical stagger to avoid overlap
        const verticalOffset = 10 + (index % 5) * 70;
        watchDiv.style.bottom = `${verticalOffset}px`;

        // Create image with fallback and tooltip
        const img = document.createElement("img");
        img.src = watch["Image URL"];
        img.alt = watch["Watch Name"];
        img.title = watch["Watch Name"];
        img.onerror = () => img.src = "https://via.placeholder.com/60?text=No+Img";

        watchDiv.appendChild(img);

        watchDiv.onclick = () => showModal(watch["Watch Name"], watch["Description"]);
        container.appendChild(watchDiv);
      });
    })
    .catch(error => console.error("Error loading watches:", error));
</script>
